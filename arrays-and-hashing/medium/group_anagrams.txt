class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        vec=defaultdict(list)

        for s in strs:
            count = [0] * 26

            for c in s:
                count[ord(c) - ord("a")] +=1

            vec[tuple(count)].append(s)
        return  list(vec.values())
